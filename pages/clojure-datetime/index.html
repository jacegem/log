변경#

(defn local-date-time
  [data]
  (walk/postwalk (fn [m]
                   (if (= (class m) java.time.ZonedDateTime)
                     (jt/local-date-time m)
                     m))
                 data))

(defn zoned-date-time
  [data]
  (walk/postwalk (fn [m]
                   (if (= (class m) java.time.LocalDateTime)
                     (jt/zoned-date-time m (ZoneId/of "Asia/Seoul"))
                     m))
                 data))
